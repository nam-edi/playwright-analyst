Formats de Résultats Supportés

Formats natifs 
{
  "project_id": 1,
  "execution_name": "Build #123",
  "results": [
    {
      "test_name": "test_login_success",
      "status": "passed",
      "duration": 1.25,
      "message": "Login successful",
      "tags": ["authentication", "critical"]
    }
  ]
}
JUnit XML
<testsuites>
  <testsuite name="LoginTests" tests="1">
    <testcase 
      classname="AuthTests" 
      name="test_login_success"

    </testcase>
  </testsuite>
</testsuites>
